494 static int da850_evm_ui_expander_setup(struct i2c_client *client, unsigned gpio,
495 unsigned ngpio, void *c)
496 {
497 int sel_a, sel_b, sel_c, ret;
498
499 sel_a = gpio + 7;
500 sel_b = gpio + 6;
501 sel_c = gpio + 5;
502
503 ret = gpio_request(sel_a, "sel_a");
504 if (ret) {
505 pr_warning("Cannot open UI expander pin %d\n", sel_a);
506 goto exp_setup_sela_fail;
507 }
508
509 ret = gpio_request(sel_b, "sel_b");
510 if (ret) {
511 pr_warning("Cannot open UI expander pin %d\n", sel_b);
512 goto exp_setup_selb_fail;
513 }
514
515 ret = gpio_request(sel_c, "sel_c");
516 if (ret) {
517 pr_warning("Cannot open UI expander pin %d\n", sel_c);
518 goto exp_setup_selc_fail;
519 }
520
521 /* deselect all functionalities */
522 gpio_direction_output(sel_a, 1);
523 gpio_direction_output(sel_b, 1);
524 gpio_direction_output(sel_c, 1);
525
526 ui_card_detected = 1;
527 pr_info("DA850/OMAP-L138 EVM UI card detected\n");
528
529 da850_evm_setup_nor_nand();
530
531 da850_evm_setup_emac_rmii(sel_a);
532
533 da850_evm_setup_char_lcd(sel_a, sel_b, sel_c);
534
535 da850_evm_setup_video_port(sel_c);
536
537 return 0;
538
539 exp_setup_selc_fail:
540 gpio_free(sel_b);
541 exp_setup_selb_fail:
542 gpio_free(sel_a);
543 exp_setup_sela_fail:
544 return ret;
545 }
